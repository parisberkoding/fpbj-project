<script>

function toggleSidebar() {
  var sidebar = document.getElementById('sidebar');
  sidebar.classList.toggle('collapsed');
  localStorage.setItem('sidebarCollapsed', sidebar.classList.contains('collapsed') ? 'true' : 'false');
}

function initSidebar(userSession) {
  if (!userSession) return;

  var roleLabels = {
    USER:'Pengusul', TIM_SURVEY:'Tim Survey',
     SCM:'SCM', GM:'General Manager'
  };
  var roleColors = {
    USER:'#8DC63F', TIM_SURVEY:'#CEDF51',
     SCM:'#0B50A1', GM:'#000000'
  };

  var roleLabelEl = document.getElementById('sidebarRoleLabel');
  if (roleLabelEl) roleLabelEl.textContent = userSession.role;

  var usernameEl = document.getElementById('sidebarUsername');
  if (usernameEl) usernameEl.textContent = '@' + userSession.username;

  var fullnameEl = document.getElementById('sidebarFullname');
  if (fullnameEl) fullnameEl.textContent = userSession.nama_lengkap;

  var divisiEl = document.getElementById('sidebarDivisi');
  if (divisiEl) divisiEl.textContent = userSession.divisi || roleLabels[userSession.role] || '';

  var dot = document.querySelector('.role-dot');
  if (dot) dot.style.background = roleColors[userSession.role] || '#8DC63F';

  // Avatar initials
  var names    = userSession.nama_lengkap.split(' ');
  var initials = names.length > 1 ? names[0][0] + names[names.length-1][0] : names[0].substring(0,2);
  var avatarEl = document.getElementById('sidebarAvatar');
  if (avatarEl) avatarEl.textContent = initials.toUpperCase();

  // Show/hide role-gated items
  var gated = document.querySelectorAll('.nav-role-gate');
  for (var i = 0; i < gated.length; i++) {
    var el    = gated[i];
    var roles = el.dataset.roles ? el.dataset.roles.split(',') : [];
    el.style.display = (roles.indexOf(userSession.role) !== -1) ? '' : 'none';
  }

  // Restore collapsed state
  var isCollapsed = localStorage.getItem('sidebarCollapsed') === 'true';
  if (isCollapsed) document.getElementById('sidebar').classList.add('collapsed');
}

function navigateTo(page, el) {
  var items = document.querySelectorAll('.nav-item');
  for (var i = 0; i < items.length; i++) items[i].classList.remove('active');
  if (el) el.classList.add('active');
  if (window.loadPage) window.loadPage(page);
}

function setBadge(id, count) {
  var el = document.getElementById(id);
  if (el) el.textContent = count > 0 ? (count > 99 ? '99+' : count) : '';
}

// Expose handleLogout to window so navbar partial can call it
window.handleLogout = function() {
  if (window.APP && window.APP.WEB_APP_URL) {
    google.script.run
      .withSuccessHandler(function() {
        sessionStorage.clear();
        window.top.location.href = window.APP.WEB_APP_URL + '?page=login';
      })
      .withFailureHandler(function() {
        sessionStorage.clear();
        window.top.location.href = window.APP.WEB_APP_URL + '?page=login';
      })
      .handleLogout(window.APP.token || '');
  } else {
    sessionStorage.clear();
    window.top.location.href = window.location.href.split('?')[0] + '?page=login';
  }
};

</script>
