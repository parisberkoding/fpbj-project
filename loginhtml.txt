// URL app - diinject GAS server-side
var WEB_APP_URL = <?!= JSON.stringify(webAppUrl) ?>;

function togglePassword() {
  var inp = document.getElementById('password');
  var ico = document.getElementById('eyeIcon');
  if (inp.type === 'password') {
    inp.type = 'text';
    ico.innerHTML = '<path d="M1 8s2.5-5 7-5 7 5 7 5-2.5 5-7 5-7-5-7-5z" stroke="currentColor" stroke-width="1.3"/><line x1="1" y1="1" x2="15" y2="15" stroke="currentColor" stroke-width="1.3"/>';
  } else {
    inp.type = 'password';
    ico.innerHTML = '<path d="M1 8s2.5-5 7-5 7 5 7 5-2.5 5-7 5-7-5-7-5z" stroke="currentColor" stroke-width="1.3"/><circle cx="8" cy="8" r="2" stroke="currentColor" stroke-width="1.3"/>';
  }
}

function showAlert(msg, type) {
  var el = document.getElementById('loginAlert');
  el.style.display = 'flex';
  el.className = 'login-alert ' + type;
  document.getElementById('alertMsg').textContent = msg;
}

function setLoading(isLoading) {
  document.getElementById('loginBtn').disabled = isLoading;
  document.getElementById('btnText').textContent = isLoading ? 'Memverifikasi...' : 'Masuk';
  document.getElementById('loginSpinner').style.display = isLoading ? 'block' : 'none';
  document.getElementById('btnArrow').style.display = isLoading ? 'none' : 'block';
}

function getRoleLandingPage(role) {
  var map = {
    'ADMIN':      'dashboard',
    'USER':       'rencana-list',
    'TIM_SURVEY': 'approval-rencana',
    'SCM':        'tender-list',
    'GM':         'dashboard'
  };
  return map[role] || 'dashboard';
}

function doRedirect(url) {
  // Multiple fallback methods for GAS iframe redirect
  try { window.top.location.href = url; } catch(e1) {
    try { window.parent.location.href = url; } catch(e2) {
      try { window.location.href = url; } catch(e3) {
        // Last resort: create a link and click it
        var a = document.createElement('a');
        a.href = url; a.target = '_top';
        document.body.appendChild(a);
        a.click();
      }
    }
  }
}

function handleLogin() {
  var username = document.getElementById('username').value.trim();
  var password = document.getElementById('password').value;

  if (!username || !password) {
    showAlert('Username dan password wajib diisi.', 'error');
    return;
  }

  setLoading(true);

  // WAJIB pakai google.script.run - fetch() kena CORS block di GAS iframe
  google.script.run
    .withSuccessHandler(function(data) {
      setLoading(false);
      if (data && data.success) {
        showAlert(data.message, 'success');

        // Tentukan halaman landing berdasarkan role user
        var landingPage = 'dashboard';
        if (data.user && data.user.role) {
          landingPage = getRoleLandingPage(data.user.role);
        }

        var redirectUrl = WEB_APP_URL + '?page=' + landingPage + '&token=' + encodeURIComponent(data.token);

        // Simpan session ke sessionStorage sebagai backup
        try {
          sessionStorage.setItem('geo_token', data.token);
          if (data.user) sessionStorage.setItem('geo_user', JSON.stringify(data.user));
          sessionStorage.setItem('geo_webapp_url', WEB_APP_URL);
        } catch(e) {}

        // Redirect dengan delay dan fallback
        setTimeout(function() {
          doRedirect(redirectUrl);
          // Safety net: jika redirect gagal setelah 3 detik, coba lagi
          setTimeout(function() {
            doRedirect(redirectUrl);
          }, 3000);
        }, 500);
      } else {
        showAlert((data && data.message) || 'Login gagal. Periksa username dan password.', 'error');
        document.getElementById('password').classList.add('error');
      }
    })
    .withFailureHandler(function(err) {
      setLoading(false);
      showAlert('Server error: ' + (err && err.message ? err.message : 'Koneksi gagal. Coba lagi.'), 'error');
    })
    .handleLogin({ username: username, password: password });
}

document.addEventListener('keydown', function(e) {
  if (e.key === 'Enter') handleLogin();
});

document.getElementById('password').addEventListener('input', function() {
  this.classList.remove('error');
});
