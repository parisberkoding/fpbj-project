function initNavbar(userSession) {
  if (!userSession) return;

  var roleLabels = {
    USER:'Pengusul', TIM_SURVEY:'Tim Survey',
    SCM:'SCM', GM:'General Manager', ADMIN:'Administrator'
  };

  var names    = userSession.nama_lengkap.split(' ');
  var initials = names.length > 1 ? names[0][0] + names[names.length-1][0] : names[0].substring(0,2);
  var init     = initials.toUpperCase();

  var navAvatar = document.getElementById('navAvatar');
  if (navAvatar)   navAvatar.textContent   = init;
  var navAvatarLg = document.getElementById('navAvatarLg');
  if (navAvatarLg) navAvatarLg.textContent = init;
  var navUsername = document.getElementById('navUsername');
  if (navUsername) navUsername.textContent = userSession.username;
  var navRole = document.getElementById('navRole');
  if (navRole)     navRole.textContent     = roleLabels[userSession.role] || userSession.role;
  var navFullname = document.getElementById('navFullname');
  if (navFullname) navFullname.textContent = userSession.nama_lengkap;
  var navEmail = document.getElementById('navEmail');
  if (navEmail)    navEmail.textContent    = userSession.email || '-';
}

function setPageTitle(title) {
  var el = document.getElementById('navPageTitle');
  if (el) el.textContent = title;
}

function toggleNotifPanel() {
  var p = document.getElementById('notifPanel');
  var m = document.getElementById('profileMenu');
  if (m) m.style.display = 'none';
  if (p) p.style.display = (p.style.display === 'none' ? 'block' : 'none');
}

function toggleProfileMenu() {
  var m = document.getElementById('profileMenu');
  var p = document.getElementById('notifPanel');
  if (p) p.style.display = 'none';
  if (m) m.style.display = (m.style.display === 'none' ? 'block' : 'none');
}

function clearNotifs() {
  var dot  = document.getElementById('notifDot');
  var list = document.getElementById('notifList');
  if (dot)  dot.style.display = 'none';
  if (list) list.innerHTML = '<div class="notif-empty"><p>Tidak ada notifikasi baru</p></div>';
}

function addNotification(icon, title, desc, time, bgColor) {
  var dot  = document.getElementById('notifDot');
  var list = document.getElementById('notifList');
  if (!dot || !list) return;
  dot.style.display = 'block';

  var empty = list.querySelector('.notif-empty');
  if (empty) list.innerHTML = '';

  var item = document.createElement('div');
  item.className = 'notif-item unread';
  item.innerHTML =
    '<div class="notif-item-icon" style="background:' + (bgColor||'#EFF5FF') + '">' + icon + '</div>' +
    '<div class="notif-item-body">' +
      '<div class="notif-item-title">' + title + '</div>' +
      '<div class="notif-item-desc">' + desc + '</div>' +
      '<div class="notif-item-time">' + time + '</div>' +
    '</div>';
  list.insertBefore(item, list.firstChild);
}

function onGlobalSearch(q) {
  if (q.length < 2) return closeSearch();
  document.getElementById('searchOverlay').style.display = 'flex';
  document.getElementById('searchQuery').textContent = q;
  if (window.globalSearch) window.globalSearch(q);
}

function closeSearch() {
  var overlay  = document.getElementById('searchOverlay');
  var searchEl = document.getElementById('globalSearch');
  if (overlay)  overlay.style.display = 'none';
  if (searchEl) searchEl.value = '';
}

function toggleMobileSidebar() {
  var sidebar = document.getElementById('sidebar');
  if (sidebar) sidebar.classList.toggle('mobile-open');
}

// Close dropdowns on outside click
document.addEventListener('click', function(e) {
  var notifWrap   = document.querySelector('.navbar-notif');
  var profileWrap = document.querySelector('.navbar-profile');
  var notifPanel  = document.getElementById('notifPanel');
  var profileMenu = document.getElementById('profileMenu');
  if (notifPanel  && notifWrap   && !notifWrap.contains(e.target))   notifPanel.style.display  = 'none';
  if (profileMenu && profileWrap && !profileWrap.contains(e.target))  profileMenu.style.display = 'none';
});

// Ctrl+K / Cmd+K shortcut
document.addEventListener('keydown', function(e) {
  if ((e.metaKey || e.ctrlKey) && e.key === 'k') {
    e.preventDefault();
    var searchEl = document.getElementById('globalSearch');
    if (searchEl) searchEl.focus();
  }
  if (e.key === 'Escape') closeSearch();
});

</script>
